// cloudcover:
// less than 20% : clear
// between 20 and 80% : partly cloudy
// more than 80% : cloudy

// lifted_index:
// less than -5 : thunderstorm

// prec_amount
// from 0 to 9

// prec_type:
// snow, rain, frzr (freezing rain), icep (ice pellets), none

// temp2m: 2m relative temperature

clear : cloudcover < 20
cloudy : 80 < cloudcover
fog : 90 < rh2m + cloudcover < 60
ishower : 0 < prec_amount < 4 + prec_type rain
lightrain : 0 < prec_amount < 4 + 80 < cloudcover + prec_type rain
lightsnow : 0 < prec_amount < 4 + prec_type snow
oshower : 0 < prec_amount < 4 + 60 < cloudcover < 80 + prec_type rain
pcloudy : 20 < cloudcover < 80
rain : 4 < prec_amount + prec_type rain
rainsnow : prec_type rain / frzr
snow : 4 < prec_amount + prec_type snow
tsrain : lifted_index < -5 + 4 < prec_amount
tstorm : lifted_index < -5 + 0 < prec_amount < 4
